# Select only numeric columns
numerical_cols = df.select_dtypes(include=np.number).columns

# Step 1: Detect Outliers (using IQR)
outliers = {}
for col in numerical_cols:
    Q1 = df[col].quantile(0.25)
    Q3 = df[col].quantile(0.75)
    IQR = Q3 - Q1
    lower = Q1 - 1.5 * IQR
    upper = Q3 + 1.5 * IQR
    outliers[col] = df[(df[col] < lower) | (df[col] > upper)][col]
    print(f"{col}: {len(outliers[col])} outliers")

# Step 2: Visualize Outliers with Colored Boxplots
palette = sns.color_palette("Set2", len(numerical_cols))  # Different colors
for i, col in enumerate(numerical_cols):
    plt.figure(figsize=(6,3))
    sns.boxplot(x=df[col], color=palette[i])  # Apply color
    plt.title(f"Boxplot of {col}", fontsize=12, weight="bold")
    plt.show();

# Step 3: Handle Outliers by Capping (Clipping to limits)
for col in numerical_cols:
    Q1 = df[col].quantile(0.25)
    Q3 = df[col].quantile(0.75)
    IQR = Q3 - Q1
    lower = Q1 - 1.5 * IQR
    upper = Q3 + 1.5 * IQR
    df[col] = df[col].clip(lower, upper)
